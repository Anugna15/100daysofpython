```python
import pandas as pd
import numpy as np

# Create a DataFrame
data = {
    'Name': ['Anu', 'Raj', 'Neha', 'Arjun', 'Sana'],
    'Age': [22, 24, 21, 23, 22],
    'City': ['Hyd', 'Delhi', 'Mumbai', 'Hyd', 'Chennai'],
    'Score': [88, 92, 95, 70, 85]
}
df = pd.DataFrame(data)

# -----------------------------
# ðŸ” Inspect Data
df.head()
df.info()
df.describe()

# -----------------------------
# ðŸŽ¯ Filtering
df[df['Score'] > 85]
df[df['City'] == 'Hyd']

# -----------------------------
# âž• Column Operations
df['Pass'] = df['Score'] > 75

# -----------------------------
# ðŸ§® Statistical Operations
df['Score'].mean()
df['Score'].median()
df['Score'].std()
df['Score'].min()
df['Score'].max()
df['Score'].sum()
df['Score'].count()
df.corr(numeric_only=True)

# -----------------------------
# ðŸ“Œ Sorting
df.sort_values(by='Score', ascending=False)
df.sort_values(by='Age')

# -----------------------------
# ðŸ§¹ Data Cleaning
dirty_df = pd.DataFrame({
    'Name': ['Anu', 'Raj', None, 'Arjun'],
    'Age': [22, None, 21, 23],
    'Score': [88, 92, 95, None]
})
cleaned_df = dirty_df.fillna(0)
cleaned_df.dropna()

# -----------------------------
# ðŸ’¾ Export to CSV
df.to_csv('day3_cleaned_data.csv', index=False)
